% Draw sidebar background + separator line on every page (below header band)
\AtBeginDocument{%
  % Left edge of text area relative to page
  \setlength{\PageLeftMargin}{\dimexpr 1in + \hoffset + \oddsidemargin\relax}%

  % Sidebar width = left column width + half the column gap
  \pgfmathsetlength{\SidebarWidth}{%
    (\LeftRatioNum/(\LeftRatioNum+\RightRatioNum))*(\textwidth-\columnsep)
    + 0.5*\columnsep
  }%

  \AddToShipoutPictureBG*{%
    \begin{tikzpicture}[remember picture,overlay]
      % Sidebar background from below header to bottom of page
      \fill[SidebarBG]
        ([xshift=\PageLeftMargin,yshift=-\HeaderBandHeight]current page.north west)
        rectangle
        ([xshift=\dimexpr\PageLeftMargin+\SidebarWidth\relax]current page.south west);

      % Separator line at the right edge of the sidebar
      \draw[Line, line width=0.7pt]
        ([xshift=\dimexpr\PageLeftMargin+\SidebarWidth\relax,yshift=-\HeaderBandHeight]current page.north west)
        --
        ([xshift=\dimexpr\PageLeftMargin+\SidebarWidth\relax]current page.south west);
    \end{tikzpicture}%
  }%
}

% -------------------- Helpers --------------------
\newcommand{\cvname}[1]{\fontsize{28}{32}\selectfont\textbf{#1}}

\newcommand{\cvtitlebox}[1]{%
  \tikz[baseline] \node[
    fill=black,
    text=white,
    inner xsep=6pt,
    inner ysep=3pt,
    font=\small\bfseries
  ] {#1};%
}

  \vspace{6pt}%

}

% Rating dots: #1 filled (0..5), #2 total (usually 5)
\newcommand{\ratingdots}[2]{%
  \begin{tikzpicture}[baseline=-0.6ex]
    \pgfmathtruncatemacro{\filled}{#1}
    \pgfmathtruncatemacro{\total}{#2}
    \foreach \i in {1,...,\total} {
      \ifnum\i>\filled
        \fill[BarBG] (0.28*\i,0) circle (1.2pt);
      \else
        \fill[Accent] (0.28*\i,0) circle (1.2pt);
      \fi
    }
  \end{tikzpicture}%
}

% Skill bar: #1 label, #2 percent (0..100)
\newcommand{\skillbar}[2]{%
  \textbf{#1}\par
  \begin{tikzpicture}
    \pgfmathsetmacro{\w}{0.05 * #2} % 100 -> 5.0 units
    \fill[BarBG] (0,0) rectangle (5,0.18);
    \fill[Accent] (0,0) rectangle (\w,0.18);
  \end{tikzpicture}\par
}

% Tag/pill
\newcommand{\pill}[1]{%
  \tikz[baseline] \node[
    draw=Line,
    fill=white,
    rounded corners=10pt,
    inner xsep=8pt,
    inner ysep=3pt,
    font=\small
  ] {#1};%
}

% Timeline entry:
% #1 years, #2 title, #3 role/area, #4 location, #5 description
\newcommand{\cvevent}[5]{%
  \begin{tabularx}{\linewidth}{@{}p{2.6cm}X@{}}
    {\color{Muted}\small #1} &
    {\textbf{#2}\par
     {\small\textbf{\color{Accent}#3} \textbullet\ \color{Muted}#4}\par
     {\small #5}} \\
  \end{tabularx}\par\vspace{6pt}
}

\newcommand{\makeheader}{%
  \begin{tikzpicture}[remember picture,overlay]
    \fill[HeaderBG]
      (current page.north west) rectangle
      ([yshift=-\HeaderBandHeight]current page.north east);

    % Put header text aligned to the text margin (no reserved photo area)
    \node[
      anchor=north west,
      align=left,
      xshift=\PageLeftMargin,
      yshift=-1.0cm
    ] {%
      \color{white}\cvname{JACK SPARROW}\par\vspace{6pt}%
      \cvtitlebox{Captain}\hspace{6pt}\cvtitlebox{Pirate}%
    };
  \end{tikzpicture}

  % Push content below the header band
  \vspace*{\dimexpr\HeaderBandHeight-\HeaderBottomGap\relax}
}
