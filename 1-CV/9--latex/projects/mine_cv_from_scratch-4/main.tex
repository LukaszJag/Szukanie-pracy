\newcommand{\rating}[2]{%
  \begingroup
  \count0=1
  \loop
    \ifnum\count0>#2
    \else
      \ifnum\count0>#1
        {\color{BarBG}\textbullet}%
      \else
        {\color{Accent}\textbullet}%
      \fi
      \hspace{2pt}%
      \advance\count0 by 1
  \repeat
  \endgroup
}
